import { Steps, Tabs, TabItem, Aside } from '@astrojs/starlight/components'

<Steps>

1. **Install packages**

   ```bash
   pnpm add @fluenti/core @fluenti/react @fluenti/next
   pnpm add -D @fluenti/cli
   ```

2. **Configure Next.js**

   ```typescript
   // next.config.ts
   import { withFluenti } from '@fluenti/next'

   export default withFluenti({
     locales: ['en', 'ja', 'zh'],
     catalogDir: './locales',
     format: 'po',
   })({
     reactStrictMode: true,
   })
   ```

   This adds the Fluenti webpack/turbopack loader and enables auto-compilation of catalogs in dev mode.

3. **Create `fluenti.config.ts`**

   ```typescript
   // fluenti.config.ts
   export default {
     sourceLocale: 'en',
     locales: ['en', 'ja', 'zh'],
     catalogDir: './locales',
     format: 'po',
     include: ['./src/**/*.{tsx,ts,jsx}'],
     compileOutDir: './src/locales/compiled',
   }
   ```

4. **Set up server-side i18n**

   ```typescript
   // src/lib/i18n.server.ts
   import { createServerI18n } from '@fluenti/react/server'

   export const { setLocale, getI18n, Trans, Plural, DateTime, NumberFormat } = createServerI18n({
     loadMessages: async (locale: string) => {
       switch (locale) {
         case 'ja': return import('@/locales/compiled/ja')
         case 'zh': return import('@/locales/compiled/zh')
         default: return import('@/locales/compiled/en')
       }
     },
     fallbackLocale: 'en',
     resolveLocale: async () => {
       const { cookies } = await import('next/headers')
       return (await cookies()).get('locale')?.value ?? 'en'
     },
   })
   ```

5. **Set locale in your root layout**

   ```tsx
   // src/app/layout.tsx
   import { setLocale } from '@/lib/i18n.server'
   import { cookies } from 'next/headers'

   export default async function RootLayout({ children }: { children: React.ReactNode }) {
     const locale = (await cookies()).get('locale')?.value ?? 'en'
     setLocale(locale)

     return (
       <html lang={locale}>
         <body>{children}</body>
       </html>
     )
   }
   ```

6. **Write a page with translations**

   <Tabs>
     <TabItem label="Server Component">
       ```tsx
       // src/app/page.tsx
       import { getI18n, Trans, Plural } from '@/lib/i18n.server'

       export default async function Home() {
         const { t } = await getI18n()

         return (
           <div>
             <h1>{t('Welcome to my app')}</h1>
             <p>{t('Hello, {name}!', { name: 'World' })}</p>

             <Trans>Read the <a href="/docs">documentation</a> for details.</Trans>
             <Plural value={5} one="# item" other="# items" />
           </div>
         )
       }
       ```
     </TabItem>
     <TabItem label="Client Component">
       ```tsx
       // src/components/SearchBar.tsx
       'use client'
       import { useI18n } from '@fluenti/react'

       export function SearchBar() {
         const { i18n } = useI18n()

         return (
           <input placeholder={i18n.t('Search...')} />
         )
       }
       ```
     </TabItem>
   </Tabs>

7. **Extract, translate, and compile**

   ```bash
   npx fluenti extract    # → locales/ja.po, locales/zh.po
   # ... translate the .po files ...
   npx fluenti compile    # → src/locales/compiled/
   ```

   <Aside type="tip">
     In dev mode, `@fluenti/next` watches your catalog directory and re-compiles automatically — you only need to run `fluenti compile` manually for production builds.
   </Aside>

</Steps>

## What `@fluenti/next` provides

| Feature | Without plugin | With plugin |
|---------|---------------|-------------|
| `t`` ` `` tagged templates | Not available | Compiled to `__i18n.t()` at build time |
| `t()` shorthand | Not available | Auto-injected `useI18n` hook |
| Catalog compilation | Manual `fluenti compile` | Auto-compile on file change in dev |
| Turbopack support | N/A | Loader registered via `turbopack.rules` |

## Server Components vs Client Components

| | Server Components (RSC) | Client Components |
|---|---|---|
| Import from | `@/lib/i18n.server` | `@fluenti/react` |
| i18n access | `await getI18n()` | `useI18n()` hook |
| Components | `Trans`, `Plural` from server setup | `Trans`, `Plural` from `@fluenti/react` |
| Rendering | Async — runs on server only | Hydrated on client |
| `t`` ` `` transform | Not needed (use `t()` from `getI18n()`) | Works with the plugin |

<Aside type="caution">
  The `t`` ` `` tagged template transform only works in **client components** (files with `'use client'`). Server Components should use `const { t } = await getI18n()` directly.
</Aside>

## Using with `<I18nProvider>` (Client Components)

To use client-side translations and locale switching, wrap your app in an `<I18nProvider>`:

```tsx
// src/components/I18nClientProvider.tsx
'use client'
import { I18nProvider } from '@fluenti/react'
import en from '@/locales/compiled/en'

export function I18nClientProvider({ locale, children }: { locale: string; children: React.ReactNode }) {
  return (
    <I18nProvider
      locale={locale}
      fallbackLocale="en"
      messages={{ en }}
      loadMessages={(loc) => import(`@/locales/compiled/${loc}`)}
    >
      {children}
    </I18nProvider>
  )
}
```

```tsx
// src/app/layout.tsx
import { setLocale } from '@/lib/i18n.server'
import { I18nClientProvider } from '@/components/I18nClientProvider'
import { cookies } from 'next/headers'

export default async function RootLayout({ children }: { children: React.ReactNode }) {
  const locale = (await cookies()).get('locale')?.value ?? 'en'
  setLocale(locale)

  return (
    <html lang={locale}>
      <body>
        <I18nClientProvider locale={locale}>
          {children}
        </I18nClientProvider>
      </body>
    </html>
  )
}
```

## Production build

```json
{
  "scripts": {
    "dev": "next dev",
    "build": "fluenti compile && next build"
  }
}
```

<Aside type="note">
  `fluenti compile` is still needed before `next build` for production. The auto-compile feature only runs in `next dev`.
</Aside>
