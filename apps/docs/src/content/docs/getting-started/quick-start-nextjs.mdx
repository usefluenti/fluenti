import { Steps, Tabs, TabItem, Aside } from '@astrojs/starlight/components'

<Steps>

1. **Install packages**

   ```bash
   pnpm add @fluenti/core @fluenti/react @fluenti/next
   pnpm add -D @fluenti/cli
   ```

2. **Configure Next.js**

   ```typescript
   // next.config.ts
   import { withFluenti } from '@fluenti/next'

   export default withFluenti({
     locales: ['en', 'ja', 'zh'],
     catalogDir: './locales',
     format: 'po',
     serverModule: './src/lib/i18n.server',
   })({
     reactStrictMode: true,
   })
   ```

   `serverModule` enables `t``\`\`` in Server Components — the loader automatically transforms them using the server i18n singleton.

3. **Create `fluenti.config.ts`**

   ```typescript
   // fluenti.config.ts
   export default {
     sourceLocale: 'en',
     locales: ['en', 'ja', 'zh'],
     catalogDir: './locales',
     format: 'po',
     include: ['./src/**/*.{tsx,ts,jsx}'],
     compileOutDir: './src/locales/compiled',
   }
   ```

4. **Set up server-side i18n**

   ```typescript
   // src/lib/i18n.server.ts
   import { configureServerI18n } from '@fluenti/next/server'

   export const { setLocale, getI18n, Trans, Plural, DateTime, NumberFormat } = configureServerI18n({
     loadMessages: async (locale: string) => {
       switch (locale) {
         case 'ja': return import('@/locales/compiled/ja')
         case 'zh': return import('@/locales/compiled/zh')
         default: return import('@/locales/compiled/en')
       }
     },
     fallbackLocale: 'en',
     resolveLocale: async () => {
       const { cookies } = await import('next/headers')
       return (await cookies()).get('locale')?.value ?? 'en'
     },
   })
   ```

   <Aside type="tip">
     Use `configureServerI18n` from `@fluenti/next/server` (not `createServerI18n` from `@fluenti/react/server`). It makes `setLocale` async to pre-load messages, enabling zero-await `t``\`\`` in Server Components.
   </Aside>

5. **Set locale in your root layout**

   ```tsx
   // src/app/layout.tsx
   import { setLocale } from '@/lib/i18n.server'
   import { cookies } from 'next/headers'

   export default async function RootLayout({ children }: { children: React.ReactNode }) {
     const locale = (await cookies()).get('locale')?.value ?? 'en'
     await setLocale(locale)

     return (
       <html lang={locale}>
         <body>{children}</body>
       </html>
     )
   }
   ```

   `await setLocale()` pre-loads messages into the per-request cache. After this, all child Server Components can access translations synchronously.

6. **Write a page with translations**

   <Tabs>
     <TabItem label="Server Component">
       ```tsx
       // src/app/page.tsx — zero imports, zero await
       export default function Home() {
         const name = 'World'

         return (
           <div>
             <h1>{t`Welcome to my app`}</h1>
             <p>{t`Hello, ${name}!`}</p>

             <Trans>Read the <a href="/docs">documentation</a> for details.</Trans>
             <Plural value={5} one="# item" other="# items" />
           </div>
         )
       }
       ```

       Zero imports needed. The loader auto-injects everything at build time:
       - `t``\`\`` → `__getServerI18n().t(...)` (synchronous call into pre-loaded cache)
       - `Trans`, `Plural`, `DateTime`, `NumberFormat` → auto-imported from `@fluenti/next/server`
     </TabItem>
     <TabItem label="Client Component">
       ```tsx
       // src/components/SearchBar.tsx
       'use client'

       export function SearchBar() {
         return (
           <input placeholder={t`Search...`} />
         )
       }
       ```

       In client components, `t``\`\`` is transformed to `__i18n.t(...)` with an auto-injected `useI18n()` hook.
     </TabItem>
   </Tabs>

7. **Extract, translate, and compile**

   ```bash
   npx fluenti extract    # → locales/ja.po, locales/zh.po
   # ... translate the .po files ...
   npx fluenti compile    # → src/locales/compiled/
   ```

   <Aside type="tip">
     In dev mode, `@fluenti/next` watches your catalog directory and re-compiles automatically — you only need to run `fluenti compile` manually for production builds.
   </Aside>

</Steps>

## What `@fluenti/next` provides

| Feature | Without plugin | With plugin |
|---------|---------------|-------------|
| `t``\`\`` in Server Components | Not available | Compiled to `__getServerI18n().t()` — sync, zero await |
| `t``\`\`` in Client Components | Not available | Compiled to `__i18n.t()` with auto-injected hook |
| Catalog compilation | Manual `fluenti compile` | Auto-compile on file change in dev |
| Turbopack support | N/A | Loader registered via `turbopack.rules` |

## Server Components vs Client Components

| | Server Components (RSC) | Client Components |
|---|---|---|
| How `t``\`\`` works | → `__getServerI18n().t()` (sync, from per-request cache) | → `__i18n.t()` (from React Context via hook) |
| Components | `Trans`, `Plural` from `@/lib/i18n.server` | `Trans`, `Plural` from `@fluenti/react` |
| Rendering | Runs on server only — zero client JS | Hydrated on client |
| Prerequisite | `await setLocale()` in layout | `<I18nProvider>` wrapper |

Both modes are fully automatic — write `t``\`\`` anywhere, the loader picks the right strategy based on whether the file has `'use client'`.

## Using with `<I18nProvider>` (Client Components)

To use client-side translations and locale switching, wrap your app in an `<I18nProvider>`:

```tsx
// src/components/I18nClientProvider.tsx
'use client'
import { I18nProvider } from '@fluenti/react'
import en from '@/locales/compiled/en'

export function I18nClientProvider({ locale, children }: { locale: string; children: React.ReactNode }) {
  return (
    <I18nProvider
      locale={locale}
      fallbackLocale="en"
      messages={{ en }}
      loadMessages={(loc) => import(`@/locales/compiled/${loc}`)}
    >
      {children}
    </I18nProvider>
  )
}
```

```tsx
// src/app/layout.tsx
import { setLocale } from '@/lib/i18n.server'
import { I18nClientProvider } from '@/components/I18nClientProvider'
import { cookies } from 'next/headers'

export default async function RootLayout({ children }: { children: React.ReactNode }) {
  const locale = (await cookies()).get('locale')?.value ?? 'en'
  await setLocale(locale)

  return (
    <html lang={locale}>
      <body>
        <I18nClientProvider locale={locale}>
          {children}
        </I18nClientProvider>
      </body>
    </html>
  )
}
```

## Production build

```json
{
  "scripts": {
    "dev": "next dev",
    "build": "fluenti compile && next build"
  }
}
```

<Aside type="note">
  `fluenti compile` is still needed before `next build` for production. The auto-compile feature only runs in `next dev`.
</Aside>
